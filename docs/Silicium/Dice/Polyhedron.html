def initialize(sides)
  @csides = 1
  @sides = [1]
  if sides.class == Integer and sides > 1
    @csides = sides
    (2..sides).each {|i| @sides << i}
  elsif sides.class == Array and sides.size > 0
    @csides = sides.size
    @sides = sides.sort
  end
end
